# Remove AppX packages for all existing users
Get-AppxPackage -AllUsers |
Where-Object {
    $_.Name -notmatch "Microsoft.DesktopAppInstaller|Microsoft.StorePurchaseApp|Microsoft.WindowsStore|Microsoft.VCLibs|Microsoft.UI.Xaml|Microsoft.NET.Native|Microsoft.SecHealthUI"
} |
ForEach-Object {
    try {
        Remove-AppxPackage -Package $_.PackageFullName -AllUsers -ErrorAction Stop
        Write-Host "Removed: $($_.Name)"
    }
    catch {
        Write-Warning "Failed: $($_.Name)"
    }
}
